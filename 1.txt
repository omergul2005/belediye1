    if (!isLoggedIn()) {
        return false;
    }
    
    $user_role = isset($_SESSION['role']) ? $_SESSION['role'] : '';
    
    if ($required_role === 'admin' && $user_role === 'admin') {
        return true;
    } elseif ($required_role === 'yonetici' && in_array($user_role, array('admin', 'yonetici'))) {
        return true;
